CREATE TABLE property_bookings (
    booking_id INT PRIMARY KEY AUTO_INCREMENT,
    property_id INT NOT NULL,
    person_id INT NOT NULL,
    
    -- Rental type classification
    rental_type ENUM('short_term', 'commercial', 'monthly') NOT NULL COMMENT '
        short_term = hourly rentals, 
        commercial = daily rentals, 
        monthly = long-term rentals',
    
    -- Rental period
    check_in DATETIME NOT NULL,
    check_out DATETIME NOT NULL,
    
    -- Automatic duration calculation
    duration_hours DECIMAL(10,2) GENERATED ALWAYS AS (
        TIMESTAMPDIFF(HOUR, check_in, check_out)
    ) STORED,
    duration_days INT GENERATED ALWAYS AS (
        DATEDIFF(check_out, check_in)
    ) STORED,
    
    -- Pricing
    base_rate DECIMAL(10,2) NOT NULL,
    rate_unit ENUM('per_hour', 'per_day', 'per_month') NOT NULL,
    cleaning_fee DECIMAL(10,2) DEFAULT 0.00,
    security_deposit DECIMAL(10,2) DEFAULT 0.00,
    total_amount DECIMAL(12,2) GENERATED ALWAYS AS (
        CASE rate_unit
            WHEN 'per_hour' THEN base_rate * duration_hours
            WHEN 'per_day' THEN base_rate * duration_days
            WHEN 'per_month' THEN base_rate * CEILING(duration_days/30)
        END + cleaning_fee + security_deposit
    ) STORED,
    
    -- Status tracking
    booking_status ENUM('pending', 'confirmed', 'checked_in', 'checked_out', 'cancelled') DEFAULT 'pending',
    is_active BOOLEAN DEFAULT TRUE,
    cancellation_reason TEXT,
    
    -- Timestamps
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    -- Foreign keys
    FOREIGN KEY (property_id) REFERENCES propertyTable(property_id),
    FOREIGN KEY (person_id) REFERENCES person(pid),
    
    -- Indexes
    INDEX idx_property_dates (property_id, check_in, check_out),
    INDEX idx_person_bookings (person_id, check_in),
    INDEX idx_booking_status (booking_status, is_active)
);

CREATE VIEW property_search_view AS
SELECT 
    p.property_id,
    p.name AS property_name,
    p.type AS property_type,
    p.description,
    p.address,
    p.zipcode,
    p.city,
    p.state_province,
    p.country,
    p.year_built,
    p.size_sqr_ft,
    p.number_of_floors,
    p.units,
    p.bedrooms,
    p.bathrooms,
    p.kitchen,
    p.living_room,
    p.furnished,
    p.parking,
    p.parking_slots,
    p.type_of_parking,
    p.security_features,
    p.utilities_included,
    p.additional_utilities,
    p.additional_amenities,
    p.purpose,
    p.rental_period,
    p.rental_price,
    p.available_date,
    p.owner_name,
    p.owner_email,
    p.owner_phone,
    p.special_instructions,
    p.legal_details,
    p.status AS property_status,
    p.person_id AS owner_id,
    p.agent_id,
    
    -- Available status based on bookings
    CASE 
        WHEN pb.booking_id IS NULL THEN 'Available'
        WHEN pb.booking_status = 'cancelled' THEN 'Available'
        WHEN pb.is_active = FALSE THEN 'Available'
        ELSE 'Booked'
    END AS availability_status,
    
    -- Current booking details (if any)
    pb.booking_id,
    pb.rental_type,
    pb.check_in,
    pb.check_out,
    pb.duration_days,
    pb.booking_status,
    pb.is_active AS booking_active,
    
    -- Person who booked (if any)
    pb.person_id AS booked_by_id
    
FROM 
    propertyTable p
LEFT JOIN 
    property_bookings pb ON p.property_id = pb.property_id
    AND (
        -- Only consider bookings that are active and not cancelled
        pb.is_active = TRUE AND
        pb.booking_status IN ('pending', 'confirmed', 'checked_in')
    );

FOR EACH ROW 
BEGIN
    DECLARE user_type VARCHAR(20);
    
    -- Determine user type
    CASE NEW.user_lvl
        WHEN 0 THEN SET user_type = 'Customer';
        WHEN 1 THEN SET user_type = 'Property Owner';
        WHEN 2 THEN SET user_type = 'Service Provider';
        WHEN 3 THEN SET user_type = 'Agent';
        WHEN 4 THEN SET user_type = 'Manager';
        ELSE SET user_type = 'User';
    END CASE;
    
    -- Account Status Change Notifications
    IF NEW.AccountStatus != OLD.AccountStatus THEN
        -- Approved Update (4)
        IF NEW.AccountStatus = 4 THEN
            INSERT INTO notifications (user_id, title, message, color)
            VALUES (
                NEW.pid,
                'Update Approved',
                CONCAT('Your ', user_type, ' profile changes have been approved.'),
                'Notification_green'
            );
            
        -- Pending Update (2)
        ELSEIF NEW.AccountStatus = 2 THEN
            -- Notification to admins

            SELECT 
                pid,
                'Update Request Pending',
                CONCAT(user_type, ' ', NEW.fname, ' ', NEW.lname, ' has requested profile changes.'),
                'Notification_orange'
            FROM person WHERE user_lvl = 4;
            
        -- Rejected Update (3)
        ELSEIF NEW.AccountStatus = 3 THEN
            INSERT INTO notifications (user_id, title, message, color)
            VALUES (
                NEW.pid,
                'Update Rejected',
                CONCAT('Your ', user_type, ' profile changes were rejected. Please contact support.'),
                'Notification_red'
            );
            
        -- Account Blocked (-1)
        ELSEIF NEW.AccountStatus = -1 THEN
            INSERT INTO notifications (user_id, title, message, color)
            VALUES (
                NEW.pid,
                'Account Blocked',
                'Your account has been temporarily blocked. Contact support for details.',
                'Notification_red'
            );
            
        -- Deletion Pending (-2)
        ELSEIF NEW.AccountStatus = -2 THEN
            -- Notification to admins
            INSERT INTO notifications (user_id, title, message, color)
            SELECT 
                pid,
                'Deletion Request',
                CONCAT(user_type, ' ', NEW.fname, ' ', NEW.lname, ' requested account deletion.'),
                'Notification_orange'
            FROM person WHERE user_lvl = 4;
            
        -- Deletion Rejected (-3)
        ELSEIF NEW.AccountStatus = -3 THEN
            INSERT INTO notifications (user_id, title, message, color)
            VALUES (
                NEW.pid,
                'Deletion Rejected',
                'Your account deletion request was rejected.',
                'Notification_blue'
            );
            
        -- Account Activated (1)
        ELSEIF NEW.AccountStatus = 1 THEN
            INSERT INTO notifications (user_id, title, message, color)
            VALUES (
                NEW.pid,
                'Account Activated',
                CONCAT('Your ', user_type, ' account is now active.'),
                'Notification_green'
            );
        END IF;
    END IF;
    
    -- Password Change Notification
    IF NEW.password != OLD.password THEN
        INSERT INTO notifications (user_id, title, message, color)
        VALUES (
            NEW.pid,
            'Password Changed',
            'Your password was updated successfully.',
            'Notification_blue'
        );

    -- Profile Update Notification (when not part of status change)
    IF (NEW.fname != OLD.fname OR NEW.lname != OLD.lname OR 
        NEW.email != OLD.email OR NEW.contact != OLD.contact) AND
        NEW.AccountStatus != 2 THEN
        INSERT INTO notifications (user_id, title, message, color)
        VALUES (
            NEW.pid,
            'Profile Updated',
            'Your profile information was updated.',
            'Notification_blue'
        );
    END IF;
    
    -- User Level Change Notification
    IF NEW.user_lvl != OLD.user_lvl THEN
        INSERT INTO notifications (user_id, title, message, color)
        VALUES (
            NEW.pid,
            'Role Changed',
            CONCAT('Your account type has been changed to ', user_type, '.'),
            'Notification_blue'

        -- Also notify admin

            CONCAT(user_type, ' ', NEW.fname, ' ', NEW.lname, ' has been assigned a new role.'),
            'Notification_orange'
        FROM person WHERE user_lvl = 4 LIMIT 1;
    END IF;

CREATE OR REPLACE VIEW Agent AS
SELECT 
    p.*,
    pd.card_id AS card_id,
    pd.card_name AS card_name,
    pd.account_no AS account_no,
    pd.bank AS bank,
    pd.branch AS branch,
    DATE_FORMAT(ap.created_at, '%Y-%m') AS assign_month,
    COUNT(CASE WHEN prop.status != 'Inactive' THEN ap.property_id END) AS property_count
FROM person p 
LEFT JOIN payment_details pd
    ON p.pid = pd.pid
LEFT JOIN agent_property ap 
    ON p.pid = ap.agent_id
LEFT JOIN propertyTable prop
    ON ap.property_id = prop.property_id
WHERE p.user_lvl = 3
GROUP BY p.pid, assign_month;
